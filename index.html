fibonacci.c
#include <stdio.h>

// 递归方法计算斐波那契数列
int fibonacci_recursive(int n) {
    if (n <= 1)
        return n;
    else
        return fibonacci_recursive(n-1) + fibonacci_recursive(n-2);
}

// 迭代方法计算斐波那契数列
int fibonacci_iterative(int n) {
    int a = 0, b = 1, c;
    if (n == 0)
        return a;
    for (int i = 2; i <= n; i++) {
        c = a + b;
        a = b;
        b = c;
    }
    return b;
}

int main() {
    int n = 10;
    
    printf("斐波那契数列的第 %d 项（递归方法）: %d\n", n, fibonacci_recursive(n));
    printf("斐波那契数列的第 %d 项（迭代方法）: %d\n", n, fibonacci_iterative(n));
    
    // 打印斐波那契数列前n项
    printf("斐波那契数列前 %d 项: ", n);
    for (int i = 0; i < n; i++) {
        printf("%d ", fibonacci_iterative(i));
    }
    printf("\n");
    
    return 0;
}

代码解释

这个程序展示了两种计算斐波那契数列的方法：递归和迭代。

  • fibonacci_recursive() - 使用递归方法计算斐波那契数列
  • fibonacci_iterative() - 使用迭代方法计算斐波那契数列
  • 递归方法简洁但效率较低，迭代方法效率更高
  • 斐波那契数列：0, 1, 1, 2, 3, 5, 8, 13, ...
sorting.c
#include <stdio.h>

// 交换两个整数的值
void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

// 冒泡排序
void bubble_sort(int arr[], int n) {
    for (int i = 0; i < n-1; i++) {
        for (int j = 0; j < n-i-1; j++) {
            if (arr[j] > arr[j+1]) {
                swap(&arr[j], &arr[j+1]);
            }
        }
    }
}

// 选择排序
void selection_sort(int arr[], int n) {
    for (int i = 0; i < n-1; i++) {
        int min_idx = i;
        for (int j = i+1; j < n; j++) {
            if (arr[j] < arr[min_idx]) {
                min_idx = j;
            }
        }
        if (min_idx != i) {
            swap(&arr[min_idx], &arr[i]);
        }
    }
}

// 打印数组
void print_array(int arr[], int n) {
    for (int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
}

int main() {
    int arr[] = {64, 34, 25, 12, 22, 11, 90};
    int n = sizeof(arr)/sizeof(arr[0]);
    
    printf("原始数组: ");
    print_array(arr, n);
    
    // 使用冒泡排序
    int arr1[n];
    for (int i = 0; i < n; i++) {
        arr1[i] = arr[i];
    }
    bubble_sort(arr1, n);
    printf("冒泡排序后的数组: ");
    print_array(arr1, n);
    
    // 使用选择排序
    int arr2[n];
    for (int i = 0; i < n; i++) {
        arr2[i] = arr[i];
    }
    selection_sort(arr2, n);
    printf("选择排序后的数组: ");
    print_array(arr2, n);
    
    return 0;
}

代码解释

这个程序展示了两种基本的排序算法：冒泡排序和选择排序。

  • bubble_sort() - 比较相邻元素并交换，重复此过程直到整个数组有序
  • selection_sort() - 每次从未排序部分选择最小元素，放到已排序部分的末尾
  • swap() - 使用指针交换两个变量的值
  • 两种排序算法的时间复杂度都是O(n²)
pointers.c
#include <stdio.h>

int main() {
    int num = 42;
    int *ptr;  // 声明一个整型指针
    
    // 指针赋值
    ptr = #  // ptr 指向 num 的地址
    
    printf("num 的值: %d\n", num);
    printf("num 的地址: %p\n", &num);
    printf("ptr 的值(存储的地址): %p\n", ptr);
    printf("ptr 指向的值: %d\n", *ptr);  // 解引用
    
    // 通过指针修改值
    *ptr = 100;
    printf("修改后 num 的值: %d\n", num);
    
    // 指针算术
    int arr[] = {10, 20, 30, 40, 50};
    int *arr_ptr = arr;  // 数组名是数组首元素的地址
    
    printf("\n数组元素通过指针访问:\n");
    for (int i = 0; i < 5; i++) {
        printf("arr[%d] = %d (地址: %p)\n", i, *(arr_ptr + i), arr_ptr + i);
    }
    
    // 指针和数组的关系
    printf("\n指针和数组的关系:\n");
    printf("arr[2] = %d\n", arr[2]);
    printf("*(arr + 2) = %d\n", *(arr + 2));
    printf("2[arr] = %d\n", 2[arr]);  // 等价于 *(2 + arr)
    
    return 0;
}

代码解释

这个程序展示了C语言中指针的基本概念和用法。

  • & - 取地址运算符
  • * - 解引用运算符
  • 指针算术：指针可以进行加减运算
  • 数组名本质上是指向数组首元素的常量指针
  • 指针和数组访问的等价性：arr[i] 等价于 *(arr + i)
file_io.c
#include <stdio.h>
#include <stdlib.h>

#define MAX_LENGTH 100

int main() {
    FILE *fp;
    char filename[50];
    char content[MAX_LENGTH];
    
    // 获取文件名
    printf("请输入文件名: ");
    scanf("%s", filename);
    
    // 写入文件
    fp = fopen(filename, "w");
    if (fp == NULL) {
        printf("无法打开文件！\n");
        return 1;
    }
    
    printf("请输入要写入的内容 (输入 'EOF' 结束):\n");
    while (1) {
        fgets(content, MAX_LENGTH, stdin);
        
        // 忽略第一次读取的空行
        if (content[0] == '\n') {
            continue;
        }
        
        // 检查是否输入EOF
        if (content[0] == 'E' && content[1] == 'O' && content[2] == 'F') {
            break;
        }
        
        fputs(content, fp);
    }
    
    fclose(fp);
    printf("文件写入成功！\n");
    
    // 读取文件
    fp = fopen(filename, "r");
    if (fp == NULL) {
        printf("无法打开文件！\n");
        return 1;
    }
    
    printf("\n文件内容如下:\n");
    while (fgets(content, MAX_LENGTH, fp) != NULL) {
        printf("%s", content);
    }
    
    fclose(fp);
    
    // 追加内容
    fp = fopen(filename, "a");
    if (fp == NULL) {
        printf("无法打开文件！\n");
        return 1;
    }
    
    fprintf(fp, "\n\n这是追加的内容！");
    fclose(fp);
    
    printf("\n内容追加成功！\n");
    
    return 0;
}

代码解释

这个程序展示了C语言中基本的文件操作：写入、读取和追加。

  • fopen() - 打开文件，使用不同模式（"w"、"r"、"a"）
  • fgets() - 从文件读取一行
  • fputs() - 写入字符串到文件
  • fprintf() - 格式化输出到文件
  • fclose() - 关闭文件
